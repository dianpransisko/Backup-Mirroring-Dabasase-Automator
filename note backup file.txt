HAK AKSES ------------------------------------------------------------------------

-- 1. Buat user khusus backup
CREATE USER user_backup WITH PASSWORD '1233x';

-- 2. Berikan izin koneksi ke database tertentu
GRANT CONNECT ON DATABASE database_penjulalan TO user_backup;

-- 3. Berikan izin penggunaan schema (biasanya public)
GRANT USAGE ON SCHEMA public TO user_backup;

-- 4. Berikan hak akses SELECT pada semua tabel saat ini
GRANT SELECT ON ALL TABLES IN SCHEMA public TO user_backup;

-- 5. Pastikan tabel yang dibuat di masa depan juga bisa dibaca
ALTER DEFAULT PRIVILEGES IN SCHEMA public GRANT SELECT ON TABLES TO user_backup;

CHEK AKSES ------------------------------------------------------------------------

SELECT 
    has_database_privilege('user_backup', 'database_penjulalan', 'CONNECT') AS bisa_konek;

----------------
Cek Hak Akses Tabel
----------------
SELECT 
    table_schema, 
    table_name, 
    privilege_type 
FROM 
    information_schema.table_privileges 
WHERE 
    grantee = 'user_backup';
